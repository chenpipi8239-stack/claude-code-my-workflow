\section{Spatial Equilibrium: Rosen-Roback}

\begin{frame}{The Core Idea}

    \begin{exampleblock}{Rosen (1979), Roback (1982)}
        If workers are \key{mobile}, then in equilibrium, utility must be equalized across all populated locations. Differences in wages must be \key{compensated} by differences in rents and amenities.
    \end{exampleblock}

    \vspace{0.5em}
    This is the spatial analog of the law of one price:
    \begin{itemize}
        \item In goods markets: arbitrage equalizes prices
        \item In spatial markets: \key{migration} equalizes utility
    \end{itemize}

    \vspace{0.5em}
    \muted{``Compensating differentials'' --- a high wage compensates for bad amenities.}

\end{frame}

\begin{frame}{Model Setup: Two Regions}

    \begin{itemize}
        \item Two regions: $i \in \{A, B\}$
        \item A continuum of \key{mobile} workers with total mass $\bar{L}$
        \item Each region has:
            \begin{itemize}
                \item Productivity: $A_i$ (exogenous)
                \item Amenity: $B_i$ (exogenous)
                \item Fixed housing supply: $H_i$
            \end{itemize}
        \item Workers choose where to live to maximize utility
    \end{itemize}

    \vspace{0.5em}
    \begin{block}{Worker Utility}
        \vspace{-0.5em}
        \[ V_i = B_i \cdot w_i^\alpha \cdot r_i^{-(1-\alpha)} \]
        where $w_i$ = wage, $r_i$ = rent, $\alpha \in (0,1)$ = expenditure share on goods.
    \end{block}

\end{frame}

\begin{frame}{Production and Housing}

    \textbf{Firms:} Competitive, produce using labor with productivity $A_i$:
    \[ w_i = A_i \]

    \vspace{0.5em}
    \textbf{Housing:} Fixed supply $H_i$, all workers must live somewhere:
    \[ L_i = H_i / h \quad \Rightarrow \quad r_i = r(L_i) \text{ increasing in } L_i \]

    For tractability, let housing market clearing imply:
    \[ r_i = \kappa \cdot L_i / H_i \]
    where $\kappa > 0$ is a housing cost parameter.

    \vspace{0.5em}
    \textbf{Population:} $L_A + L_B = \bar{L}$

\end{frame}

\begin{frame}{Spatial Equilibrium Condition}

    Mobile workers choose location to maximize $V_i$. In equilibrium:

    \vspace{0.5em}
    \begin{block}{Indifference Condition}
        \vspace{-0.5em}
        \[ V_A = V_B \quad \Longleftrightarrow \quad B_A \cdot w_A^\alpha \cdot r_A^{-(1-\alpha)} = B_B \cdot w_B^\alpha \cdot r_B^{-(1-\alpha)} \]
    \end{block}

    \vspace{0.5em}
    Combined with:
    \begin{itemize}
        \item $w_i = A_i$ (labor demand)
        \item $r_i = \kappa L_i / H_i$ (housing market)
        \item $L_A + L_B = \bar{L}$ (population)
    \end{itemize}

    \vspace{0.5em}
    \key{Three equations, three unknowns} ($L_A$, $r_A$, $r_B$). The system pins down the spatial distribution of population.

\end{frame}

\begin{frame}{Solving: A Worked Example}

    \textbf{Setup:} $A_A = A_B = 1$, $H_A = H_B = 1$, $\bar{L} = 2$, $\alpha = 0.5$, $\kappa = 1$.

    \vspace{0.3em}
    \textbf{Case 1:} $B_A = B_B = 1$ (symmetric amenities)

    By symmetry: $L_A = L_B = 1$, $r_A = r_B = 1$, $V_A = V_B = 1$.

    \vspace{0.3em}
    \textbf{Case 2:} $B_A = 2$, $B_B = 1$ (Region A more attractive)

    From $V_A = V_B$:
    \[ 2 \cdot 1^{0.5} \cdot r_A^{-0.5} = 1 \cdot 1^{0.5} \cdot r_B^{-0.5} \]

    With $r_i = L_i$ and $L_A + L_B = 2$:
    \[ 2 \cdot L_A^{-0.5} = (2 - L_A)^{-0.5} \]
    \[ \Rightarrow \quad L_A \approx 1.6, \quad L_B \approx 0.4 \]

    \vspace{0.3em}
    \key{Result:} The attractive region has \positive{more people} but \negative{higher rents}. This is the compensating differential.

\end{frame}

\begin{frame}{Rosen-Roback Diagram}

    \begin{center}
    \begin{tikzpicture}[scale=0.85]

        \draw[-{Stealth}, thick] (0,0) -- (6.5,0) node[right] {$L_A$};

            node[right, font=\small] {$V_B(\bar{L} - L_A)$};

    \vspace{0.3em}
    \muted{Equilibrium: where $V_A(L_A) = V_B(\bar{L} - L_A)$. Higher amenities in $A$ shift $V_A$ up, attracting more workers until rents restore indifference.}

\end{frame}

\begin{frame}{Comparative Statics}

    What happens when Region A's \key{productivity} increases ($A_A \uparrow$)?

    \vspace{0.5em}
    \begin{enumerate}
        \item $w_A$ rises $\Rightarrow$ $V_A > V_B$
        \item Workers migrate $A \leftarrow B$
        \item $L_A$ increases, $r_A$ rises
        \item $L_B$ decreases, $r_B$ falls
        \item New equilibrium: $V_A = V_B$ again
    \end{enumerate}

    \vspace{0.5em}
    \begin{block}{Key Insight}
        In spatial equilibrium, a productivity shock in one region affects \key{all} regions through migration. Local shocks have \key{general equilibrium} effects.
    \end{block}

\end{frame}

\begin{frame}{Limitations of Rosen-Roback}

    The basic Rosen-Roback model is powerful but limited:

    \vspace{0.5em}
    \begin{itemize}
        \item \negative{No trade} --- regions are autarkic
        \item \negative{No variety} --- a single homogeneous good
        \item \negative{No micro-foundation} for why productivity differs
        \item \negative{Static} --- no dynamics, no transition paths
        \item \negative{Frictionless migration} --- unrealistic for short run
    \end{itemize}

    \vspace{0.5em}
    We address these one by one:
    \begin{itemize}
        \item \key{Trade + variety:} CES preferences (next)
        \item \key{Productivity micro-foundation:} Eaton-Kortum (Lecture 3)
        \item \key{Dynamics:} CDP 2019 (Lectures 5--6)
    \end{itemize}

\end{frame}

\begin{frame}{Socratic Check}

    \begin{block}{Question for the Class}
        In the Rosen-Roback model, if Region A receives a large positive amenity shock, what happens to \key{welfare} in Region B?
    \end{block}

    \vspace{1em}
    Think about:
    \begin{itemize}
        \item Workers migrate from B to A
        \item Rents in B fall
        \item In equilibrium, $V_A = V_B = \bar{V}$
        \item But is $\bar{V}$ higher or lower than before?
    \end{itemize}

    \vspace{0.5em}
    \muted{Answer: $\bar{V}$ increases --- workers in B benefit from the amenity improvement in A through lower rents. Spatial equilibrium transmits gains.}

\end{frame}

\section{From Preferences to Price Indices}

\begin{frame}{Why Do We Need CES?}

    The Rosen-Roback model has one good. But in reality:
    \begin{itemize}
        \item Regions produce \key{differentiated goods}
        \item Consumers value \key{variety}
        \item Trade patterns depend on \key{substitutability}
    \end{itemize}

    \vspace{0.5em}
    \begin{exampleblock}{CES = Constant Elasticity of Substitution}
        A family of utility functions where the \key{elasticity of substitution} $\sigma$ between any two goods is constant. This gives us:
        \begin{itemize}
            \item Tractable price indices
            \item Clean expenditure shares
            \item A single parameter $\sigma$ governing substitution
        \end{itemize}
    \end{exampleblock}

\end{frame}

\begin{frame}{CES Utility}

    A consumer in region $j$ has preferences over goods from all regions:

    \vspace{0.3em}
    \begin{block}{CES Utility Function}
        \vspace{-0.5em}
        \[ U_j = \left( \sum_{i=1}^{N} \left( c_{ij} \right)^{\frac{\sigma-1}{\sigma}} \right)^{\frac{\sigma}{\sigma-1}}, \qquad \sigma > 1 \]
    \end{block}

    \vspace{0.3em}
    where:
    \begin{itemize}
        \item $c_{ij}$ = consumption in $j$ of goods from region $i$
        \item $\sigma$ = elasticity of substitution between goods
        \item $N$ = number of regions
    \end{itemize}

    \vspace{0.3em}
    \key{Interpretation:} $\sigma$ measures how willing consumers are to substitute between goods from different regions. Higher $\sigma$ $\Rightarrow$ goods are more substitutable.

\end{frame}

\begin{frame}{Special Cases of $\sigma$}

    \begin{center}
    \begin{tikzpicture}[scale=0.75]
        % Perfect complements
        \begin{scope}[shift={(0,0)}]
            \draw[-{Stealth}] (0,0) -- (3.5,0) node[right, font=\tiny] {$c_1$};
            \draw[-{Stealth}] (0,0) -- (0,3.5) node[above, font=\tiny] {$c_2$};
            \draw[thick, tsinghua] (0.5,2.5) -- (0.5,0.5) -- (2.5,0.5);
            \draw[thick, tsinghua] (1,3) -- (1,1) -- (3,1);
            \node[below, font=\small] at (1.75, -0.3) {$\sigma \to 0$};
            \node[below, font=\tiny] at (1.75, -0.7) {Perfect complements};
        \end{scope}
        % Cobb-Douglas
        \begin{scope}[shift={(5,0)}]
            \draw[-{Stealth}] (0,0) -- (3.5,0) node[right, font=\tiny] {$c_1$};
            \draw[-{Stealth}] (0,0) -- (0,3.5) node[above, font=\tiny] {$c_2$};
            \draw[thick, tsinghua] plot[smooth, domain=0.3:3] (\x, {0.8/\x});

            \node[below, font=\small] at (1.75, -0.3) {$\sigma = 1$};

        \end{scope}
        % Perfect substitutes
        \begin{scope}[shift={(10,0)}]

            \draw[thick, tsinghua] (0.3,2.5) -- (2.5,0.3);
            \draw[thick, tsinghua] (0.5,3.2) -- (3.2,0.5);
            \node[below, font=\small] at (1.75, -0.3) {$\sigma \to \infty$};

        \end{scope}
    \end{tikzpicture}
    \end{center}

    \vspace{0.3em}
    In trade models, $\sigma > 1$ (goods are substitutes, but not perfect). Typical estimates: $\sigma \in [3, 10]$ depending on the level of aggregation.

\end{frame}

\begin{frame}{Demand and the Price Index}

    Consumer in $j$ maximizes $U_j$ subject to $\sum_i p_{ij} c_{ij} = w_j$.

    \vspace{0.3em}
    \textbf{Optimal demand:}
    \[ c_{ij} = \left(\frac{p_{ij}}{P_j}\right)^{-\sigma} \frac{w_j}{P_j} \]

    \begin{block}{CES Price Index}
        \vspace{-0.5em}
        \[ P_j = \left( \sum_{i=1}^{N} p_{ij}^{1-\sigma} \right)^{\frac{1}{1-\sigma}} \]
    \end{block}

    \vspace{0.2em}
    \key{Properties:} $P_j$ aggregates all prices facing consumers in $j$. Lower prices or more variety $\Rightarrow$ lower $P_j$ $\Rightarrow$ higher real income.

\end{frame}

\begin{frame}{Expenditure Shares}

    The share of $j$'s total expenditure spent on goods from region $i$:

    \vspace{0.3em}
    \begin{block}{Expenditure / Trade Shares}
        \vspace{-0.5em}
        \[ \pi_{ij} = \frac{p_{ij}^{1-\sigma}}{\sum_{k=1}^{N} p_{kj}^{1-\sigma}} = \frac{p_{ij}^{1-\sigma}}{P_j^{1-\sigma}} \]
    \end{block}

    \vspace{0.3em}
    \key{Key properties:}
    \begin{itemize}
        \item $\sum_{i} \pi_{ij} = 1$ for each $j$ (exhaustive)
        \item $\pi_{ij}$ increasing in $p_{kj}$ for $k \neq i$ (if competitors are expensive, $i$ gains share)
        \item $\pi_{ij}$ decreasing in $p_{ij}$ (higher price $\Rightarrow$ lower share)
        \item Higher $\sigma$ $\Rightarrow$ shares more sensitive to price differences
    \end{itemize}

    \vspace{0.3em}
    \muted{These $\pi_{ij}$ will become the ``trade shares'' once we add trade costs.}

\end{frame}

\begin{frame}{Worked Example: Two Goods}

    \textbf{Setup:} $N=2$, prices $p_1 = 1$, $p_2 = 2$.

    \vspace{0.3em}
    Expenditure share on good 1:
    \[ \pi_1 = \frac{p_1^{1-\sigma}}{p_1^{1-\sigma} + p_2^{1-\sigma}} = \frac{1}{1 + 2^{1-\sigma}} \]

    \vspace{0.3em}
    {\small
    \begin{center}
    \begin{tabular}{ccc}
        \toprule
        $\sigma$ & $\pi_1$ (cheap good) & $\pi_2$ (expensive good) \\
        \midrule
        $1$ (Cobb-Douglas) & 0.50 & 0.50 \\
        $2$ & 0.67 & 0.33 \\
        $5$ & 0.94 & 0.06 \\
        $10$ & 0.998 & 0.002 \\
        \bottomrule
    \end{tabular}
    \end{center}
    }

    \vspace{0.3em}
    \key{Takeaway:} As $\sigma$ increases, consumers shift spending toward the cheaper good. At $\sigma=1$, spending shares equal income shares regardless of prices.

\end{frame}

\begin{frame}{Real Wages}

    The \key{real wage} in region $i$ is what matters for welfare:

    \vspace{0.3em}
    \begin{block}{Real Wage}
        \vspace{-0.5em}
        \[ \omega_i = \frac{w_i}{P_i} \]
    \end{block}

    \vspace{0.5em}
    \begin{itemize}
        \item A worker earning \$100k in Beijing with $P_{\text{Beijing}} = 2$ has real wage $\omega = 50$
        \item A worker earning \$50k in Guiyang with $P_{\text{Guiyang}} = 0.8$ has real wage $\omega = 62.5$
    \end{itemize}

    \vspace{0.5em}
    In spatial equilibrium with CES preferences:
    \[ B_i \cdot \omega_i = B_i \cdot \frac{w_i}{P_i} = \bar{U} \quad \text{for all } i \]

    \vspace{0.3em}
    \muted{The price index $P_i$ connects trade patterns to welfare.}

\end{frame}

\section{Trade Costs and the Gravity of Trade}

\begin{frame}{Why Trade Costs?}

    So far, we have assumed consumers can access all goods frictionlessly. In reality:

    \vspace{0.5em}
    \begin{itemize}
        \item \key{Transport costs:} shipping, logistics, infrastructure
        \item \key{Tariffs and trade barriers:} policy-imposed frictions
        \item \key{Information frictions:} finding partners, verifying quality
        \item \key{Cultural/language barriers:} contract enforcement, trust
    \end{itemize}

    \vspace{0.5em}
    These costs are \key{large}: Anderson \& van Wincoop (2004) estimate that trade costs for industrialized countries are equivalent to a 170\% ad-valorem tariff.

    \vspace{0.3em}
    \key{Trade costs are the reason geography matters for economics.}

\end{frame}

\begin{frame}{Iceberg Trade Costs}

    \begin{exampleblock}{Samuelson (1954) ``Iceberg'' Formulation}
        To deliver 1 unit of a good from region $i$ to region $j$, the exporter must ship $d_{ij} \geq 1$ units. The fraction $d_{ij} - 1$ ``melts'' in transit.
    \end{exampleblock}

    \vspace{0.5em}
    \key{Conventions:}
    \begin{itemize}
        \item $d_{ii} = 1$ (no cost to trade with yourself)
        \item $d_{ij} \geq 1$ for $i \neq j$
        \item Not necessarily symmetric: $d_{ij} \neq d_{ji}$ in general
    \end{itemize}

    \vspace{0.5em}
    \key{Why ``iceberg''?}
    \begin{itemize}
        \item Multiplicative: price in $j$ of goods from $i$ is $p_{ij} = p_i \cdot d_{ij}$
        \item No separate transport sector needed
        \item Analytically clean: trade costs enter like a ``tax'' on prices
    \end{itemize}

\end{frame}

    \begin{center}

    \muted{Regions 2 and 3 are ``far apart'' ($d_{23}=2.0$). Region 1 is a hub.}

\end{frame}

\begin{frame}{Trade Shares with Trade Costs}

    Combining CES preferences with iceberg trade costs:

    \vspace{0.3em}
    \begin{block}{CES Trade Shares}
        \vspace{-0.5em}
        \[ \pi_{ij} = \frac{(p_i \cdot d_{ij})^{1-\sigma}}{\displaystyle\sum_{k=1}^{N} (p_k \cdot d_{kj})^{1-\sigma}} \]
    \end{block}

    \vspace{0.2em}
    \key{Reading the formula:}
    \begin{itemize}
        \item $\pi_{ij}$ = fraction of $j$'s spending on goods from $i$
        \item Numerator: competitiveness of $i$ in market $j$
        \item Denominator: total competitiveness of \emph{all} suppliers
        \item Higher $d_{ij}$ $\Rightarrow$ lower $\pi_{ij}$; higher $\sigma$ $\Rightarrow$ trade costs matter more
    \end{itemize}

\end{frame}

\begin{frame}{The Gravity Equation Emerges}

    Total trade from $i$ to $j$:
    \[ X_{ij} = \pi_{ij} \cdot E_j \]
    where $E_j = w_j L_j$ is total expenditure in $j$.

    \vspace{0.3em}
    Substituting the trade share:
    \[ X_{ij} = \frac{(p_i \cdot d_{ij})^{1-\sigma}}{P_j^{1-\sigma}} \cdot E_j \]

    \vspace{0.3em}
    This has the \key{gravity structure}:
    \begin{itemize}
        \item $X_{ij}$ increases with the ``size'' of the exporter ($p_i^{1-\sigma}$ reflects supply capacity)
        \item $X_{ij}$ increases with the ``size'' of the importer ($E_j$)
        \item $X_{ij}$ decreases with trade costs ($d_{ij}^{1-\sigma}$)
        \item $X_{ij}$ depends on \key{multilateral resistance} ($P_j$) --- Anderson \& van Wincoop (2003)
    \end{itemize}

\end{frame}

\begin{frame}{Worked Example: 3 Regions with Trade Costs}

    \textbf{Setup:} $N=3$, $p_i = 1$ for all $i$, $\sigma = 5$.

    Trade costs from previous diagram: $d_{12}=d_{21}=1.3$, $d_{13}=d_{31}=1.1$, $d_{23}=d_{32}=2.0$.

    \vspace{0.3em}
    Trade share of region 1's spending on its own goods:
    \[ \pi_{11} = \frac{1^{-4}}{1^{-4} + 1.3^{-4} + 1.1^{-4}} = \frac{1}{1 + 0.350 + 0.683} = \key{0.492} \]

    \vspace{0.3em}
    {\small
    \begin{center}
    \begin{tabular}{c|ccc}
        $\pi_{ij}$ & $j=1$ & $j=2$ & $j=3$ \\
        \hline
        $i=1$ & \key{0.492} & 0.190 & 0.365 \\
        $i=2$ & 0.172 & \key{0.542} & 0.035 \\
        $i=3$ & 0.336 & 0.268 & \key{0.600} \\
    \end{tabular}
    \end{center}
    }

    \vspace{0.2em}
    \muted{Each column sums to 1. Region 2 buys very little from region 3 (high $d_{23}$). Region 1 (the hub) trades with both.}

\end{frame}

\begin{frame}{Properties of Trade Shares}

    \begin{enumerate}
        \item \key{Exhaustive:} $\sum_{i=1}^{N} \pi_{ij} = 1$ for each $j$
        \item \key{Home bias:} $\pi_{ii} > \pi_{ij}$ when $d_{ij} > 1$ (people buy more from home)
        \item \key{Distance decay:} $\pi_{ij}$ decreasing in $d_{ij}$
        \item \key{Trade elasticity:} $\frac{\partial \ln \pi_{ij}}{\partial \ln d_{ij}} \approx 1 - \sigma < 0$
        \item \key{Multilateral resistance:} $\pi_{ij}$ depends not just on $d_{ij}$ but on \emph{all} $d_{kj}$
    \end{enumerate}

    \vspace{0.5em}
    \begin{exampleblock}{Property 5 is crucial}
        If region $j$ becomes more remote from \emph{all} other regions, its home share $\pi_{jj}$ rises even if $d_{ij}$ doesn't change. Trade shares are about \key{relative} costs.
    \end{exampleblock}

\end{frame}

\begin{frame}{Market Clearing and Trade Balance}

    Total income in region $i$ equals total spending on $i$'s goods:

    \vspace{0.3em}
    \begin{block}{Market Clearing}
        \vspace{-0.5em}
        \[ w_i L_i = \sum_{j=1}^{N} \pi_{ij} \cdot w_j L_j \]
    \end{block}

    \vspace{0.3em}
    \key{Reading this:}
    \begin{itemize}
        \item LHS: total income (= total production) in region $i$
        \item RHS: sum of what each region $j$ spends on $i$'s goods
        \item This is a \key{system of $N$ equations} in wages $(w_1, \ldots, w_N)$
        \item One equation is redundant (Walras' law) $\Rightarrow$ normalize one wage
    \end{itemize}

    \vspace{0.3em}
    \muted{With trade shares depending on wages through prices, this is a fixed-point problem.}

\end{frame}

\section{A Simple Spatial General Equilibrium}

\begin{frame}{Combining the Building Blocks}

    We now have all the ingredients:

    \vspace{0.3em}
    \begin{enumerate}
        \item \key{Spatial equilibrium:} Workers choose locations (utility equalized)
        \item \key{CES preferences:} Consumers value variety, prices aggregate to $P_j$
        \item \key{Trade costs:} Geography matters ($d_{ij}$)
        \item \key{Trade shares:} $\pi_{ij}$ determine who buys from whom
        \item \key{Market clearing:} Income = sales
    \end{enumerate}

    \vspace{0.5em}
    Putting them together gives us a \key{simple spatial general equilibrium model}. This is the prototype for everything in Lectures 3--8.

\end{frame}

\begin{frame}{The System of Equations}

    \begin{block}{Simple Spatial GE ($N$ regions)}
    {\footnotesize
    \begin{align}
        \text{Trade shares:} \quad & \pi_{ij} = \frac{(w_i d_{ij})^{1-\sigma}}{\sum_k (w_k d_{kj})^{1-\sigma}} \label{eq:shares} \\[3pt]
        \text{Price index:} \quad & P_j = \left(\sum_i (w_i d_{ij})^{1-\sigma}\right)^{\frac{1}{1-\sigma}} \label{eq:price} \\[3pt]
        \text{Market clearing:} \quad & w_i L_i = \sum_j \pi_{ij} \cdot w_j L_j \label{eq:mc} \\[3pt]
        \text{Spatial eq.:} \quad & B_i \cdot \frac{w_i}{P_i} = \bar{U} \quad \forall i \label{eq:spatial} \\[3pt]
        \text{Population:} \quad & \sum_i L_i = \bar{L} \label{eq:pop}
    \end{align}
    }
    \end{block}

    \muted{Unknowns: $\{w_i, L_i, P_i, \pi_{ij}\}$. Given: $\{A_i, B_i, d_{ij}, \bar{L}, \sigma\}$.}

\end{frame}

\begin{frame}{What This System Does}

    The system simultaneously determines:

    \vspace{0.3em}
    \begin{itemize}
        \item \key{Where people live} ($L_i$) --- through spatial equilibrium
        \item \key{What they earn} ($w_i$) --- through market clearing
        \item \key{What they pay} ($P_i$) --- through trade shares and trade costs
        \item \key{Who trades with whom} ($\pi_{ij}$) --- through CES + trade costs
    \end{itemize}

    \vspace{0.5em}
    \begin{block}{Circular Causation}
        More people in $i$ $\Rightarrow$ more production $\Rightarrow$ lower prices for $i$'s goods $\Rightarrow$ higher real wages nearby $\Rightarrow$ attracts more people. But also: more people $\Rightarrow$ higher rents $\Rightarrow$ lower real wages $\Rightarrow$ pushes people away.
    \end{block}

    \vspace{0.3em}
    \muted{Whether agglomeration or dispersion wins depends on parameters.}

\end{frame}

\begin{frame}{Existence and Uniqueness}

    Does this system always have a solution? Is it unique?

    \vspace{0.5em}
    \begin{itemize}
        \item \key{Existence:} Yes, under mild regularity conditions (continuous mapping theorem / Brouwer's fixed point)
        \item \key{Uniqueness:} \negative{Not guaranteed} in general
            \begin{itemize}
                \item Multiple equilibria are possible when agglomeration forces are strong
                \item Small changes in fundamentals can lead to large spatial reorganization
            \end{itemize}
        \item \key{In practice:} Uniqueness holds for ``most'' parameter values in quantitative work. Numerical methods find stable equilibria.
    \end{itemize}

    \vspace{0.5em}
    \muted{We will discuss computational approaches (fixed-point iteration, dampening) in Lecture 8.}

\end{frame}

\begin{frame}{Welfare}

    Welfare of a worker in region $i$ is the \key{real wage} adjusted for amenities:

    \[ W_i = B_i \cdot \frac{w_i}{P_i} \]

    In spatial equilibrium: $W_i = \bar{U}$ for all populated regions.

    \vspace{0.2em}
    \begin{exampleblock}{Arkolakis, Costinot \& Rodriguez-Clare (2012)}
        Welfare gains from trade can be summarized by:
        $W_i^{\text{trade}} / W_i^{\text{autarky}} = \pi_{ii}^{-1/(\sigma-1)}$.
        Only two statistics needed: home trade share $\pi_{ii}$ and trade elasticity $\sigma - 1$.
    \end{exampleblock}

    \lecmark{3}

\end{frame}

\begin{frame}{What We Cannot Do Yet}

    This simple model is a good starting point, but:

    \vspace{0.3em}
    \begin{itemize}
        \item \negative{No micro-foundation for productivity:} Why is $A_i$ different? $\Rightarrow$ \key{EK (Lecture 3)}
        \item \negative{No dynamics:} How do workers transition? $\Rightarrow$ \key{CDP (Lectures 5--6)}
        \item \negative{No growth:} How does the economy evolve? $\Rightarrow$ \key{DNR, CCPX (Lectures 4, 7)}
        \item \negative{No computation:} How to solve counterfactuals? $\Rightarrow$ \key{Hat algebra (Lecture 3)}
    \end{itemize}

    \vspace{0.5em}
    \begin{block}{Next Lecture}
        Lecture 3 adds the \key{Eaton-Kortum} supply side (Frechet productivity) and introduces \key{exact hat algebra} --- the key computational tool.
    \end{block}

\end{frame}

\begin{frame}{Summary}

    \begin{enumerate}
        \item \key{Rosen-Roback:} Spatial equilibrium equalizes utility across locations. Wages compensate for rents and amenities.
        \item \key{CES preferences:} Tractable aggregation with elasticity $\sigma$. Gives us price indices and expenditure shares.
        \item \key{Iceberg trade costs:} $d_{ij} \geq 1$ makes geography matter. Prices in $j$ reflect origin price $\times$ trade cost.
        \item \key{Trade shares:} $\pi_{ij}$ captures who buys from whom. Gravity structure emerges naturally.
        \item \key{Spatial GE:} Five-equation system determines wages, prices, population, and trade simultaneously.
    \end{enumerate}

    \vspace{0.5em}
    \key{Key notation introduced:} $w_i$, $r_i$, $L_i$, $P_j$, $\sigma$, $\pi_{ij}$, $d_{ij}$, $\omega_i = w_i/P_i$

\end{frame}
